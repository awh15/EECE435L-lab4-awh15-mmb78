def validate_name(name):
    if len(name.strip()) == 0:
        return False
    if not name.isalpha():
        return False
    return True

    if not age.isdigit() or not (1 <= int(age) <= 100):

    def add_student(self):

    def edit_record(self):
        selected_row = self.records_table.currentRow()
        if selected_row == -1:
            QMessageBox.warning(self, "Error", "Please select a record to edit.")
            return

        record_type = self.records_table.item(selected_row, 0).text()
        record_id = self.records_table.item(selected_row, 1).text()

        dialog = QDialog(self)
        dialog.setWindowTitle(f"Edit {record_type}")
        layout = QFormLayout()

        if record_type == "Student":
            student = get_student_by_id(record_id) 

            name_input = QLineEdit(dialog)
            name_input.setText(student.name)
            layout.addRow("Name:", name_input)

            age_input = QLineEdit(dialog)
            age_input.setText(str(student.age))
            layout.addRow("Age:", age_input)

            email_input = QLineEdit(dialog)
            email_input.setText(student.get_email())
            layout.addRow("Email:", email_input)

            id_input = QLineEdit(dialog)
            id_input.setText(student.student_id)
            layout.addRow("ID:", id_input)

        elif record_type == "Instructor":
            instructor = get_instructor_by_id(record_id)  

            name_input = QLineEdit(dialog)
            name_input.setText(instructor.name)
            layout.addRow("Name:", name_input)

            age_input = QLineEdit(dialog)
            age_input.setText(str(instructor.age))
            layout.addRow("Age:", age_input)

            email_input = QLineEdit(dialog)
            email_input.setText(instructor.get_email())
            layout.addRow("Email:", email_input)

            id_input = QLineEdit(dialog)
            id_input.setText(instructor.instructor_id)
            layout.addRow("ID:", id_input)

        elif record_type == "Course":
            course = get_course_by_id(record_id)  

            name_input = QLineEdit(dialog)
            name_input.setText(course.course_name)
            layout.addRow("Course Name:", name_input)

            instructor_input = QLineEdit(dialog)
            instructor_input.setText(course.instructor.name)
            layout.addRow("Instructor Name:", instructor_input)

        button_box = QDialogButtonBox(QDialogButtonBox.Ok | QDialogButtonBox.Cancel)
        layout.addWidget(button_box)

        button_box.accepted.connect(dialog.accept)
        button_box.rejected.connect(dialog.reject)

        dialog.setLayout(layout)

        if dialog.exec():

            if record_type in ["Student", "Instructor"]:
                if not validate_name(name_input.text()) or not validate_age(age_input.text()) or not validate_email(email_input.text()) or not validate_id(id_input.text()):
                    QMessageBox.warning(self, "Input Error", "Invalid input. Please check the fields.")
                    return

            if record_type == "Student":
                student.name = name_input.text()
                student.age = int(age_input.text())
                student.set_email(email_input.text())
                student.student_id = id_input.text()
                update_student(student)  

            elif record_type == "Instructor":
                instructor.name = name_input.text()
                instructor.age = int(age_input.text())
                instructor.set_email(email_input.text())
                instructor.instructor_id = id_input.text()
                update_instructor(instructor)  

            elif record_type == "Course":
                if not validate_name(name_input.text()):
                    QMessageBox.warning(self, "Input Error", "Invalid course name")
                    return

                course.course_name = name_input.text()
                selected_instructor = get_instructor_by_name(instructor_input.text())  
                if selected_instructor:
                    course.instructor = selected_instructor
                    update_course(course)  
                else:
                    QMessageBox.warning(self, "Error", "Instructor not found")
                    return

            self.display_records()

    def export_to_csv(self):
        """Exports all records to a CSV file."""
        # Open a file dialog to select the save location
        options = QFileDialog.Options()
        file_path, _ = QFileDialog.getSaveFileName(self, "Save CSV", "", "CSV Files (*.csv);;All Files (*)", options=options)

        if not file_path:
            return  # User canceled the file dialog

        try:
            # Open the CSV file for writing
            with open(file_path, mode='w', newline='') as file:
                writer = csv.writer(file)

                # Write the headers for students
                writer.writerow(['Type', 'ID', 'Name', 'Age', 'Email', 'Registered Courses'])

                # Fetch all students and their courses from the database
                conn = connect()
                c = conn.cursor()
                c.execute('''
                    SELECT students.student_id, students.name, students.age, students.email, 
                        GROUP_CONCAT(courses.course_name, ', ') AS courses 
                    FROM students
                    LEFT JOIN enrollments ON students.student_id = enrollments.student_id
                    LEFT JOIN courses ON enrollments.course_id = courses.course_id
                    GROUP BY students.student_id
                ''')
                students = c.fetchall()
                for student in students:
                    writer.writerow(['Student', student[0], student[1], student[2], student[3], student[4] or "No courses"])

                # Write the headers for instructors
                writer.writerow(['Type', 'ID', 'Name', 'Age', 'Email', 'Assigned Courses'])

                # Fetch all instructors and their assigned courses from the database
                c.execute('''
                    SELECT instructors.instructor_id, instructors.name, instructors.age, instructors.email, 
                        GROUP_CONCAT(courses.course_name, ', ') AS courses
                    FROM instructors
                    LEFT JOIN courses ON instructors.instructor_id = courses.instructor_id
                    GROUP BY instructors.instructor_id
                ''')
                instructors = c.fetchall()
                for instructor in instructors:
                    writer.writerow(['Instructor', instructor[0], instructor[1], instructor[2], instructor[3], instructor[4] or "No courses"])

                # Write the headers for courses
                writer.writerow(['Type', 'Course ID', 'Course Name', 'Instructor', 'Enrolled Students'])

                # Fetch all courses and their enrolled students from the database
                c.execute('''
                    SELECT courses.course_id, courses.course_name, instructors.name AS instructor_name,
                        GROUP_CONCAT(students.name, ', ') AS students
                    FROM courses
                    LEFT JOIN instructors ON courses.instructor_id = instructors.instructor_id
                    LEFT JOIN enrollments ON courses.course_id = enrollments.course_id
                    LEFT JOIN students ON enrollments.student_id = students.student_id
                    GROUP BY courses.course_id
                ''')
                courses = c.fetchall()
                for course in courses:
                    writer.writerow(['Course', course[0], course[1], course[2], course[3] or "No students enrolled"])

                conn.close()

            QMessageBox.information(self, "Success", "Records exported successfully to CSV!")
        except Exception as e:
            QMessageBox.warning(self, "Error", f"Failed to export records: {str(e)}")







if __name__ == '__main__':
    app = QApplication(sys.argv)
    window = SchoolManagementSystem()
    window.show()
    sys.exit(app.exec_())
